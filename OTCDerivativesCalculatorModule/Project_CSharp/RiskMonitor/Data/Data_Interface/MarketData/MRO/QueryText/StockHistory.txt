SELECT  STD_DATE, SECTOR_ID , STOCK_ID , STOCK_NAME, CURR_PRICE, 
               ADJ_STOCK_PRICE, MIN_PRICE, MAX_PRICE, DIVIDEND_RATE
FROM     TD_MRSTOCKFACTOR
WHERE  (SUBSTR(STOCK_ID, 1, 3) = 'KR7') 
		AND STOCK_ID = '@stockID@' 
		AND STD_DATE > '@startDate@' AND STD_DATE < '@endDate@'
		AND (SECTOR_ID = 'KOSPI200') ORDER BY 1
